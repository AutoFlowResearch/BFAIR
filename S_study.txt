CREATE TABLE public.study
(
    identifier "char" NOT NULL,
    title "char" NOT NULL,
    description "char" NOT NULL,
    submission_date date NOT NULL,
    public_release_date date,
    publications "char",
    contacts "char",
    design_type "char",
    factor_name "char",
    factor_type "char",
    CONSTRAINT study_pkey PRIMARY KEY (identifier)
)

TABLESPACE pg_default;

ALTER TABLE public.study
    OWNER to postgres;
