CREATE TABLE public."Investigation"
(
    identifier "char" NOT NULL,
    title "char" NOT NULL,
    description "char",
    submission_date date NOT NULL,
    public_release_date date,
    contact "char",
    publication character(1) COLLATE pg_catalog."default",
    CONSTRAINT "Investigation_pkey" PRIMARY KEY (identifier)
)

TABLESPACE pg_default;
